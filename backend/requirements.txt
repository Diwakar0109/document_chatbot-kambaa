# Web Framework & Server
fastapi                 # The web framework itself
uvicorn                 # ASGI server for local development
gunicorn                # Production-grade server, essential for deployment

# Environment & Utilities
python-dotenv           # To load environment variables from .env file
python-multipart        # Required by FastAPI for handling file uploads and form data

# LangChain Core & Integrations
langchain               # Core LangChain library
langchain-community     # For community components like the FAISS wrapper
langchain-groq          # For Groq LLM integration
langchain-cohere        # For Cohere embeddings integration

# Vector Store
faiss-cpu               # The FAISS vector store library (CPU version)

# Document Parsers
pypdf                   # To parse .pdf files
python-docx             # To parse .docx files
pandas                  # To parse .xlsx and .xls files
openpyxl                # Required by Pandas for reading modern Excel (.xlsx) files
